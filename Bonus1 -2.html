def objective(w, x, y, lam):
    n, d = x.shape
    yx = numpy.multiply(y, x) # n-by-d matrix
    yxw = numpy.dot(yx, w) # n-by-1 matrix
    vec1 = numpy.exp(-yxw) # n-by-1 matrix
    vec2 = numpy.log(1 + vec1) # n-by-1 matrix
    loss = numpy.mean(vec2) # scalar
    reg = lam / 2 * numpy.sum(w * w) # scalar
    return loss + reg

class Worker:
    def __init__(self,x,y):
        self.x = x
        self.y = y
        self.s = x.shape[0]
        self.d = x.shape[1]
        self.w = numpy.zeros((self.d,1))
    def assign_neighbors(self,l):
        self.neighbor = l
        self.n = len(l)
    def get_neighbors(self):
        return self.neighbor
    def get_w(self):
        return self.w
    def loss(self):
        yx = numpy.multiply(self.y,self.x)
        yxw = numpy.dot(yx,self.w)
        vec1 = numpy.exp(-yxw)
        vec2 = numpy.log(1+vec1)
        return numpy.sum(vec2)
    def gradient(self,lam):
        yx = numpy.multiply(self.y,self.x)
        yxw = numpy.dot(yx,self.w)
        vec1 = numpy.exp(yxw)
        vec2 = numpy.divide(yx,1+vec1)
        vec3 = -numpy.mean(vec2, axis=0).reshape(self.d, 1)
        self.g = vec3+ lam * self.w
        #self.g = vec2
    def gd(self,alpha):
        self.w = self.w_ - alpha*self.g
        
    def weighted_average(self,neighbor_w,my_weight=0.5):
        other_weights = (1-my_weight)/self.n
        self.w_ = my_weight*self.w
        for i in range(self.n):
            self.w_ += other_weights*neighbor_w[i]
    def average(self,neighbor_w):
        self.w_=self.w
        for w in neighbor_w:
            self.w_ += w
        self.w_ /= (self.n+1)

import math

def create_workers(m,x,y):
    n,d = x.shape
    s = math.floor(n/m)
    workers = []
    
    for i in range(m):
        indices = list(range(i*s,(i+1)*s))
        worker = Worker(x[indices,:],y[indices,:])
        workers.append(worker)
        
    return workers

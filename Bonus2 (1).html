PCA and the standard autoencoder are unsupervised dimensionality reduction methods, and their learned features are not discriminative. If you build a classifier upon the low-dimenional features extracted by PCA and autoencoder, you will find the classification accuracy very poor.

Linear discriminant analysis (LDA) is a traditionally supervised dimensionality reduction method for learning low-dimensional features which are highly discriminative. Likewise, can we extend autoencoder to supervised leanring?

import numpy

def to_one_hot(y, num_class=10):
    results = numpy.zeros((len(y), num_class))
    for i, label in enumerate(y):
        results[i, label] = 1.
    return results

from keras.layers import Dense, Input
from keras import models

input_img = Input(shape=(784,), name='input_img')

encode1 = Dense(128, activation='relu', name='encode1')(input_img)
encode2 = Dense(32, activation='relu', name='encode2')(encode1)
encode3 = Dense(8, activation='relu', name='encode3')(encode2)
bottleneck = Dense(2, activation='relu', name='bottleneck')(encode3)
decode1 = Dense(8, activation='relu', name='decode1')(bottleneck)
decode2 = Dense(32, activation='relu', name='decode2')(decode1)
decode3 = Dense(128, activation='relu', name='decode3')(decode2)

# build the supervised autoencoder network
from keras.layers import Dense, Input, Dropout,BatchNormalization,Activation
from keras import models

input_img = Input(shape=(784,), name='input_img')

# encoder network
#encode1 = <add a dense layer taking input_img as input>
#<Add more layers...>
dropout1 = Dropout(0.5, name='dropout1')(input_img)
#encode1 = Dense(128, activation='relu', name='encode1')(dropout1)
#encode2 = Dense(32, activation='relu', name='encode2')(encode1)
#encode3 = Dense(8, activation='relu', name='encode3')(encode2)
encode1 = Dense(128, name='encode1')(dropout1)
batchnorm1 = BatchNormalization()(encode1)
encodeact1 = Activation('relu')(batchnorm1)
encode2 = Dense(32,  name='encode2')(encodeact1)
batchnorm2 = BatchNormalization()(encode2)
encodeact2 = Activation('relu')(batchnorm2)
encode3 = Dense(8,  name='encode3')(encodeact2)
batchnorm3 = BatchNormalization()(encode3)
encodeact3 = Activation('relu')(batchnorm3)


# The width of the bottleneck layer must be exactly 2.
#bottleneck = <the output of encoder network>
bottleneck = Dense(2, activation='relu', name='bottleneck')(encode3)

# decoder network
#decode1 = <add a dense layer taking bottleneck as input>
#<Add more layers...>
#decode4 = <the output of decoder network>
decode1 = Dense(8, name='decode1')(bottleneck)
batchnorm4 = BatchNormalization()(decode1)
decodeact1 = Activation('relu')(batchnorm4)
decode2 = Dense(32,  name='decode2')(decodeact1)
batchnorm5 = BatchNormalization()(decode2)
decodeact2 = Activation('relu')(batchnorm5)
decode3 = Dense(128, name='decode3')(decodeact2)
batchnorm6 = BatchNormalization()(decode3)
decodeact3 = Activation('relu')(batchnorm6)
dropout2 = Dropout(0.5, name='dropout2')(decodeact3)
decode4 = Dense(784, name='decode4')(dropout2)
batchnorm7 = BatchNormalization()(decode4)
decodeact4 = Activation('relu')(batchnorm7)

# build a classifier upon the bottleneck layer
#classifier1 = <add a dense layer taking bottleneck as input>
#<Add more dense layers and regularizations...>
#classifier3 = <the output of classifier network>
classifier1 = Dense(8, activation='relu', name='classifier1')(bottleneck)
classifier1 = Dense(32, activation='relu', name='classifier1')(bottleneck)
classifier2 = Dense(128, activation='relu', name='classifier2')(classifier1)
classifier3 = Dense(10, activation='softmax', name='classifier3')(classifier2)
